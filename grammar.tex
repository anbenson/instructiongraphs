% Frequently used commands
\newcommand{\sort}[1]{\mathtt{#1}}
\newcommand{\syn}[1]{\textbf{#1}}
\newcommand{\nterm}[1]{\textit{#1}}

% Frequently used syntax
\newcommand{\prog}[2]{\syn{P(}\nterm{#1}, \nterm{#2}\syn{)}}
\newcommand{\sing}[1]{\syn{S(}\nterm{#1}\syn{)}}
\newcommand{\cons}[2]{\nterm{#1}\syn{::}\nterm{#2}}
\newcommand{\ver}[2]{\syn{V(}\nterm{#1}, \nterm{#2}\syn{)}}
\newcommand{\doonce}[2]{\syn{do }\nterm{#1}\syn{ then }\nterm{#2}}
\newcommand{\dountil}[3]{\syn{do }\nterm{#1}\syn{ until }\nterm{#2}\syn{ then }\nterm{#3}}
\newcommand{\ifelse}[3]{\syn{if }\nterm{#1}\syn{ then }\nterm{#2}\syn{ else }\nterm{#3}}
\newcommand{\goto}[1]{\syn{goto }\nterm{#1}}
\newcommand{\progend}[0]{\syn{end}}

\begin{alignat*}{4}
  \sort{Program}\quad
    &\nterm{p}\quad & &::=\quad &
      &\prog{vs}{n}\qquad & &\text{programs}\\
  \sort{Vertices}\quad
    &\nterm{vs}\quad & &::=\quad &
      &\sing{v}\qquad & &\text{singleton}\\
  &&&&&\cons{v}{vs}\qquad & &\text{cons}\\
  \sort{Vertex}\quad
    &\nterm{v}\quad & &::=\quad &
      &\ver{n}{c}\qquad & &\text{vertex}\\
  \sort{Content}\quad
    &\nterm{c}\quad & &::=\quad &
      &\doonce{a}{n}\qquad & &\text{single action}\\
  &&&&&\dountil{a}{cnd}{n}\qquad & &\text{open loop action}\\
  &&&&&\ifelse{cnd}{n}{n}\qquad & & \text{conditional}\\
  &&&&&\goto{n}\qquad & &\text{goto}\\
  &&&&&\progend\qquad & &\text{termination}
\end{alignat*}

We let $\nterm{n} \in \mathbb{Z}$, the integers.\\

We let \nterm{a} $\in$ $\sort{Actions}$, a sort describing classes of actions,
like movement, that a robot might be able to perform. A grammar defining
$\sort{Actions}$ is assumed.\\

We let \nterm{cnd} $\in$ $\sort{Conditions}$, a sort describing classes of
conditions, like whether an object is some distance ahead, that a robot might be
able to detect. A grammar defining $\sort{Conditions}$ is assumed.\\
